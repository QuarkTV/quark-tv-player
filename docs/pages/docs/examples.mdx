# Examples

## Basic Usage

The simplest way to use QuarkTV Player:

```jsx
import { QuarkTVPlayer, BunnyStreamClient } from '@quarktv/quark-tv-player'

const client = new BunnyStreamClient({
  apiKey: 'your-api-key',
  libraryId: 'your-library-id'
})

export default function BasicPlayer() {
  return (
    <QuarkTVPlayer
      videoId="your-video-id"
      config={client}
    />
  )
}
```

## With Streamer Information

Add streamer details to enhance the live streaming experience:

```jsx
export default function StreamerPlayer() {
  return (
    <QuarkTVPlayer
      videoId="your-video-id"
      config={client}
      streamerInfo={{
        name: "VanixQC",
        title: "Just Chatting",
        category: "IRL",
        viewers: 1234,
        avatarUrl: "https://example.com/avatar.jpg"
      }}
    />
  )
}
```

## With Event Handlers

Listen to player events:

```jsx
export default function PlayerWithEvents() {
  return (
    <QuarkTVPlayer
      videoId="your-video-id"
      config={client}
      onPlay={() => {
        console.log('Video started playing')
        // Analytics tracking
      }}
      onPause={() => {
        console.log('Video paused')
        // Update UI
      }}
      onError={(error) => {
        console.error('Playback error:', error)
        // Show error message
      }}
    />
  )
}
```

## Next.js Integration

Using QuarkTV Player in a Next.js application:

```jsx
// pages/_app.tsx
import '@quarktv/quark-tv-player/dist/styles.css'

function MyApp({ Component, pageProps }) {
  return <Component {...pageProps} />
}

export default MyApp

// pages/stream.tsx
import dynamic from 'next/dynamic'

const QuarkTVPlayer = dynamic(
  () => import('@quarktv/quark-tv-player').then(mod => mod.QuarkTVPlayer),
  { ssr: false }
)

export default function StreamPage() {
  return (
    <QuarkTVPlayer
      videoId="your-video-id"
      config={client}
    />
  )
}
```

## Error Boundary

Wrap the player in an error boundary for better error handling:

```jsx
import { Component } from 'react'

class PlayerErrorBoundary extends Component {
  state = { hasError: false }

  static getDerivedStateFromError(error) {
    return { hasError: true }
  }

  componentDidCatch(error, errorInfo) {
    console.error('Player error:', error, errorInfo)
  }

  render() {
    if (this.state.hasError) {
      return (
        <div className="error-container">
          <h2>Something went wrong with the player</h2>
          <button onClick={() => this.setState({ hasError: false })}>
            Try again
          </button>
        </div>
      )
    }

    return this.props.children
  }
}

export default function PlayerWithErrorBoundary() {
  return (
    <PlayerErrorBoundary>
      <QuarkTVPlayer
        videoId="your-video-id"
        config={client}
      />
    </PlayerErrorBoundary>
  )
}
